rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {
    // Files can be read and written only by the authenticated user who owns the folder.
    // The folder structure will be 'users/{userId}/...'
    match /users/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Allow any authenticated user to upload feedback screenshots.
    // The unique feedback ID in the path ensures they can't overwrite others' files.
    match /feedback/{feedbackId}/{fileName} {
      allow write: if request.auth != null;
    }
  }
}
