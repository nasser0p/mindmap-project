rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {
    // Files can be read and written only by the authenticated user who owns the folder.
    // The folder structure will be 'users/{userId}/...'
    match /users/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
